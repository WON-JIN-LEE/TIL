# 슈퍼유저 권한을 이용해서 커맨드를 쓰려면 sudo를 붙인다.

# 시스템 패키지 업데이트
$ sudo apt update

# 시스템 패키지중 업그레이드 가능한 목록 리스트
$ sudo apt list --upgradable 

# 패키지 업그레이드 커맨드
$ sudo apt upgrade -y


# 세션 접속 시간
$ uptime


# 메모리 사용량 출력
$ free

# 현재 디스크 사용량 확인
$ df -h

# 우분투 패키지 이름 확인
$ uname -a

# 현재 폴더 경로
$ pwd

# home의 위치 경로로 이동
$ cd ~


# 파일 목록 출력
$ ls -a
$ ls -al
$ ls -alt (수정된 일시로 내림차순 정렬되어 출력)
$ ls -altr (수정된 일시로 오름차순 정렬되어 출력)

옵션
-a: 숨김 파일 포함
-l: 자세히 정보
-i: "inode 번호"를 출력
-t: 최근 수정된 파일이나 폴더 부터
-r: 역순
